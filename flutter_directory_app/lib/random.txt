import 'package:flutter/material.dart';
import 'package:image_picker/image_picker.dart';


class Person {
  PickedFile? photo;
  String name = '';
  String gotra = '';
  String occupation = '';
  String contact = '';
  String birthplace = '';
  String currentAddress = '';
}

class RegistrationPage extends StatefulWidget {
  @override
  _RegistrationPageState createState() => _RegistrationPageState();
}

class _RegistrationPageState extends State<RegistrationPage> {
  final ImagePicker _imagePicker = ImagePicker();
  Person headOfFamily = Person();
  Person wife = Person();

  Future<void> _pickImage(Person person) async {
    final pickedFile = await _imagePicker.pickImage(source: ImageSource.gallery);
    setState(() {
      person.photo = pickedFile as PickedFile?;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Family Registration'),
      ),
      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.stretch,
          children: [
            _buildPersonForm("Head of the Family", headOfFamily),
            SizedBox(height: 20),
            _buildPersonForm("Wife", wife),
            SizedBox(height: 20),
            ElevatedButton(
              onPressed: _registerNow,
              child: Text('Register Now'),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildPersonForm(String title, Person person) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.stretch,
      children: [
        Text(
          title,
          style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
        ),
        SizedBox(height: 10),
        // _buildImagePicker(person),
        _buildTextField(person.name, 'Name', (value) => person.name = value),
        _buildTextField(person.gotra, 'Gotra', (value) => person.gotra = value),
        _buildTextField(person.occupation, 'Occupation', (value) => person.occupation = value),
        _buildTextField(person.contact, 'Mobile Number or Email', (value) => person.contact = value),
        _buildTextField(person.birthplace, 'Birthplace', (value) => person.birthplace = value),
        _buildTextField(person.currentAddress, 'Current Address', (value) => person.currentAddress = value),
      ],
    );
  }

  Widget _buildImagePicker(Person person) {
    return Column(
      children: [
        person.photo == null
            ? Placeholder(
                fallbackHeight: 100,
                fallbackWidth: 100,
              )
            : Image.file(
                File(person.photo!.path),
                height: 100,
                width: 100,
              ),
        ElevatedButton(
          onPressed: () => _pickImage(person),
          child: Text('Pick Photo'),
        ),
      ],
    );
  }

  Widget _buildTextField(String value, String label, void Function(String) onChanged) {
    return Padding(
      padding: const EdgeInsets.symmetric(vertical: 8.0),
      child: TextField(
        onChanged: onChanged,
        decoration: InputDecoration(
          labelText: label,
          border: OutlineInputBorder(),
        ),
      ),
    );
  }

  void _registerNow() {
    // Process the registration data
    print("Head of the Family: ${headOfFamily.name}, Wife: ${wife.name}");
    // Add your logic to store the data or navigate to the next screen
  }
}
